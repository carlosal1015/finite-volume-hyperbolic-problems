\begin{frame}
	We attempt to solve~\eqref{eq:complicatedsystem} with the
	fractional step method~\cite{Leveque2002}
	\begin{equation}
		\left\{
		\begin{aligned}
			\diffp{\symbf{u}}{t}+\diffp*{\symbf{f}\left(\symbf{u}\right)}{x} & =
			\symbf{0}.                                                           \\
			\diff{\symbf{u}}{t}                                              & =
			\symbf{s}\left(\symbf{u}\right).
		\end{aligned}
		\right.
	\end{equation}

	\begin{columns}
		\begin{column}{.4\paperwidth}
			\begin{description}
				\item[Upwind method]

				      \begin{equation*}
					      \frac{\symbf{u}^{n+1}_{j}-\symbf{u}^{n}_{j}}{\Delta t}+
					      \frac{\symbf{f}^{n}_{j}-\symbf{f}^{n}_{j-1}}{\Delta x}=
					      \symbf{0}.\implies
					      \symbf{u}^{n+1}_{j}=
					      \symbf{u}^{n}_{j}
					      -\frac{\Delta t}{\Delta x}
					      \left(\symbf{f}^{n}_{j}-\symbf{f}^{n}_{j-1}\right).
				      \end{equation*}

				\item[Forward Euler]

				      \begin{equation*}
					      \frac{\symbf{u}^{n+1}_{j}-\symbf{u}^{n}_{j}}{\Delta t}=
					      \symbf{s}\left(\symbf{u}^{n}_{j}\right).\implies
					      \symbf{u}^{n+1}_{j}=
					      \symbf{u}^{n}_{j}+\Delta t
					      \symbf{s}\left(\symbf{u}^{n}_{j}\right).
				      \end{equation*}

				\item[Fractional step]

				      \begin{equation*}
					      \left\{
					      \begin{aligned}
						      \symbf{u}^{n+1}_{j}          & =
						      \alert{\symbf{u}^{\ast}_{j}}+
						      \Delta t
						      \symbf{s}\left(\alert{\symbf{u}^{\ast}_{j}}\right). \\
						      \alert{\symbf{u}^{\ast}_{j}} & =
						      \symbf{u}^{n}_{j}-
						      \frac{\Delta t}{\Delta x}
						      \left(\symbf{f}^{n}_{j}-\symbf{f}^{n}_{j-1}\right).
					      \end{aligned}
					      \right.
				      \end{equation*}
			\end{description}
		\end{column}
		\begin{column}{.4\paperwidth}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
	\begin{block}{Direct Finite Volume}
		\begin{equation*}
			\symbf{u}^{n+1-}_{j}=
			\symbf{u}^{n}_{j}-
			\frac{\Delta t}{\Delta x}
			\left(\symbf{g}^{n}_{j+\frac{1}{2}}-\symbf{g}^{n}_{j-\frac{1}{2}}\right).
		\end{equation*}
	\end{block}

	\begin{block}{Implicit Euler}
		\begin{equation*}
			\symbf{u}^{n+1}_{j}=
			\symbf{u}^{n+1-}_{j}+
			\Delta t
			\symbf{s}\left(\symbf{u}^{n+1}_{j}\right).
		\end{equation*}
	\end{block}

	\begin{equation*}
		\symbf{g}^{n}_{j+\frac{1}{2}}=
		\symbf{g}\left(\symbf{u}^{n}_{j},\symbf{u}^{n}_{j+1}\right),\qquad
		\symbf{u}^{0}_{j}=
		\frac{1}{\Delta x}
		\int_{C_{j}}
		\symbf{u}_{0}\left(x\right)\dl x.
	\end{equation*}

	\begin{block}{Fractional Step Methods}
		\begin{align*}
			\symbf{u}^{n+1}_{j}                                             & =
			\symbf{u}^{n}_{j}-
			\frac{\Delta t}{\Delta x}
			\left(
			\symbf{g}^{n}_{j+\frac{1}{2}}-\symbf{g}^{n}_{j-\frac{1}{2}}
			\right)+
			\Delta t
			\symbf{s}
			\left(
			\symbf{u}^{n+1}_{j}
			\right).                                                            \\
			\begin{bmatrix}
				\mathcolor{mygreen}{\theta}^{n+1}_{j}       \\
				{\mathcolor{myred}{S_{\text{y}}}}^{n+1}_{j} \\
				{\mathcolor{myblue}{S_{\text{o}}}}^{n+1}_{j}
			\end{bmatrix} & =
			\begin{bmatrix}
				\mathcolor{mygreen}{\theta}^{n}_{j}       \\
				{\mathcolor{myred}{S_{\text{y}}}}^{n}_{j} \\
				{\mathcolor{myblue}{S_{\text{o}}}}^{n}_{j}
			\end{bmatrix}-
			\frac{\Delta t}{\Delta x}
			\left(
			\symbf{g}\left(
				\begin{bmatrix}
						\mathcolor{mygreen}{\theta}^{n}_{j}       \\
						{\mathcolor{myred}{S_{\text{y}}}}^{n}_{j} \\
						{\mathcolor{myblue}{S_{\text{o}}}}^{n}_{j}
					\end{bmatrix},
				\begin{bmatrix}
						\mathcolor{mygreen}{\theta}^{n}_{j+1}       \\
						{\mathcolor{myred}{S_{\text{y}}}}^{n}_{j+1} \\
						{\mathcolor{myblue}{S_{\text{o}}}}^{n}_{j+1}
					\end{bmatrix}\right)-
			\symbf{g}
			\left(
				\begin{bmatrix}
						\mathcolor{mygreen}{\theta}^{n}_{j-1}       \\
						{\mathcolor{myred}{S_{\text{y}}}}^{n}_{j-1} \\
						{\mathcolor{myblue}{S_{\text{o}}}}^{n}_{j-1}
					\end{bmatrix},
				\begin{bmatrix}
						\mathcolor{mygreen}{\theta}^{n}_{j}       \\
						{\mathcolor{myred}{S_{\text{y}}}}^{n}_{j} \\
						{\mathcolor{myblue}{S_{\text{o}}}}^{n}_{j}
					\end{bmatrix}
				\right)
			\right)
			+\Delta t
			\symbf{s}
			\left(
			\begin{bmatrix}
					\mathcolor{mygreen}{\theta}^{n+1}_{j}       \\
					{\mathcolor{myred}{S_{\text{y}}}}^{n+1}_{j} \\
					{\mathcolor{myblue}{S_{\text{o}}}}^{n+1}_{j}
				\end{bmatrix}
			\right).
		\end{align*}
	\end{block}
