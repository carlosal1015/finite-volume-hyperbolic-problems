\begin{frame}
	\begin{columns}
		\begin{column}{0.55\textwidth}
			\begin{align}
				\shortintertext{The Crank-Nicolson scheme uses a combination of two first-order in
					time methods to produce a method that is second-order in time.}
				\frac{\symbf{u}^{n+1}_{j}-\symbf{u}^{n}_{j}}{\Delta t}+
				\frac{\symbf{f}^{n}_{j+1}-\symbf{f}^{n}_{j-1}}{2\Delta x}                                             & =
				\symbf{s}^{n}_{j}.                                                \tag{FTCS}\label{eq:FTCS}                                                    \\
				\frac{\symbf{u}^{n+1}_{j}-\symbf{u}^{n}_{j}}{\Delta t}+
				\frac{\symbf{f}^{n+1}_{j+1}-\symbf{f}^{n+1}_{j-1}}{2\Delta x}                                         & =
				\symbf{s}^{n+1}_{j}.                                                \tag{BTCS}\label{eq:BTCS}
				\shortintertext{The Crank-Nicolson scheme uses the average of equations~\eqref{eq:FTCS} and~\eqref{eq:BTCS}.}
				\frac{\symbf{u}^{n+1}_{j}-\symbf{u}^{n}_{j}}{\Delta t}+
				\frac{\symbf{f}^{n}_{j+1}-\symbf{f}^{n}_{j-1}+\symbf{f}^{n+1}_{j+1}-\symbf{f}^{n+1}_{j-1}}{4\Delta x} & =
				\frac{\symbf{s}^{n}_{j}+\symbf{s}^{n+1}_{j}}{2}.                                                                                        \notag \\
				\symbf{u}^{n+1}_{j}-\symbf{u}^{n}_{j}+
				\frac{\Delta t}{4\Delta x}
				\left(\symbf{f}^{n}_{j+1}-\symbf{f}^{n}_{j-1}+\symbf{f}^{n+1}_{j+1}-\symbf{f}^{n+1}_{j-1}\right)      & =
				\frac{\Delta t}{2}
				\left(\symbf{s}^{n}_{j}+\symbf{s}^{n+1}_{j}\right).\notag
			\end{align}
		\end{column}
		\begin{column}{0.35\textwidth}
			\begin{figure}[ht!]
				\centering
				\includegraphics[width=.35\paperwidth]{stencilcranknicolson}
				\caption{Molecular diagram of~\eqref{eq:CN} scheme.}
			\end{figure}
		\end{column}
	\end{columns}

	\

	Let $r\coloneqq\frac{\Delta t}{4\Delta x}$, which is rearranged to
	give
	\begin{equation}
		-r\symbf{f}^{n+1}_{j-1}+
		\symbf{u}^{n+1}_{j}+
		r\symbf{f}^{n+1}_{j+1}-
		\frac{\Delta t}{2}\symbf{s}^{n+1}_{j}=
		r\symbf{f}^{n}_{j-1}+
		\symbf{u}^{n}_{j}-
		r\symbf{f}^{n}_{j+1}+
		\frac{\Delta t}{2}\symbf{s}^{n}_{j}.
		\tag{CN}\label{eq:CN}
	\end{equation}
	In general, the left side of~\eqref{eq:CN} contains four unknown,
	and the right side four known, pivotal values of $\symbf{u}^{n}$.
\end{frame}
