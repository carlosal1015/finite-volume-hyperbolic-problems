El producto de Hadamard se define como
\begin{align*}
	\odot\colon\mathbb{R}^{n}\times\mathbb{R}^{n} & \longrightarrow\mathbb{R}^{n} \\
	\left(
	\begin{bmatrix}
			a_{1}  \\
			\vdots \\
			a_{n}
		\end{bmatrix},
	\begin{bmatrix}
			b_{1}  \\
			\vdots \\
			b_{n}
		\end{bmatrix}
	\right)                                       & \longmapsto
	\begin{bmatrix}
		a_{1}b_{1} \\
		\vdots     \\
		a_{n}b_{n}
	\end{bmatrix}.
\end{align*}
El producto exterior se define como
\begin{align*}
	\otimes\colon\mathbb{R}^{m}\times\mathbb{R}^{n} & \longrightarrow\mathbb{R}^{m\times n} \\
	\left(
	\begin{bmatrix}
			a_{1}  \\
			\vdots \\
			a_{m}
		\end{bmatrix},
	\begin{bmatrix}
			b_{1}  \\
			\vdots \\
			b_{n}
		\end{bmatrix}
	\right)                                         & \longmapsto
	\begin{bmatrix}
		a_{1}b_{1} & \cdots & a_{1}b_{n} \\
		\vdots     & \ddots & \vdots     \\
		a_{m}b_{1} & \cdots & a_{m}b_{n}
	\end{bmatrix}.
\end{align*}
